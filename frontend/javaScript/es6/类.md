    super(calories);
    this.flavor = flavor;
    this.toppings = toppings;
  }

    this.toppings.push(topping);
  }
}

```js
class Tree {
  constructor(size = '10', leaves = { spring: 'green', summer: 'green', fall: 'ora' }) {
    this.size = size;
    this.leaves = leaves;
    this.leafColor = null;
  }
  changeSeason(season) {
    this.leafColor = this.leaves[season];
    if (season === 'spring') {
      this.size += 1;
    }
  }
}

class Maple extends Tree {
  constructor(syrupQty = 15, size, leaves) {
    super(size, leaves);
    this.syrupQty = syrupQty;
  }
  changeSeason(season) {
    super.changeSeason(season);
    if (season === 'spring') {
      this.syrupQty += 1;
    }
  }
  gattherSyrup() {
    this.syrupQty -= 3;
  }
}

const myMaple = new Maple(15, 5);
myMaple.changeSeason('fall');
myMaple.gattherSyrup();
myMaple.changeSeason('spring');
```

#### 静态方法 static

静态方法调用直接在类上进行，不能在类的实例上调整。
静态方法通常用于创建实用程序函数。

```js
class StaticMethodCall {
  static staticMethod() {
    return '';
  }
}
```
